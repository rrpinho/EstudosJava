Crie um programa orientado a objetos, no qual o usuário preenche um vetor
de inteiros e que possua dois métodos. Um para mostrar qual a posição do
maior numero e outro para mostrar o array invertido:




package aaa;
import java.io.*;
import java.util.Scanner;
public class Aaa {
    public static void main(String[] args) {
        Scanner entrada = new Scanner(System.in);
        int [] vet = new int [5];
        mostrar vetor = new mostrar();
        System.out.println("Digite 5 numeros inteiros aleatorios");
        for(int i=0; i<vet.length; i++){
            vet[i] = entrada.nextInt();            
        }
        vetor.mostrarVetor(vet);
    }
}








package aaa;
public class mostrar {
    public void mostrarVetor(int[] vet){
            for(int i=0; i<vet.length; i++){
            System.out.println(vet[i]);
            }
        }
}

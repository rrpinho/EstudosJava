public class Conta{
	private float saldo = 0;
	public inicializador(float valor){
		this.saldo = valor;
	}


	public void saca (float valor){
		if (valor<saldo)
			this.saldo = valor;
		else
			System.out.println("Saldo Insuficiente");
	}
}
